		NOT = { tag = GER }

	}

	}
}

			is_subject = no

				factor = 25

					is_major = yes

					}
				}
			}
		}
		name = generics.15.a

			limit = {
				FROM = {
					is_subject_of = ROOT
				}
			}
			custom_effect_tooltip = FROM_is_puppet_slavic_states
			every_state = {
				limit = {
					is_controlled_by = FROM
					OR = {
						has_state_flag = slavic_states
						has_state_flag = slavic_states_czech
						has_state_flag = german_slavic_state
					}	
				}
				SOV = {
					transfer_state = PREV
				}

			FROM = {
				country_event = { id = generics.18 days = 1 }
			}
		}
		else = {
			every_state = {
				limit = {
					is_controlled_by = ROOT
					OR = {
						has_state_flag = slavic_states
						has_state_flag = slavic_states_czech
						has_state_flag = german_slavic_state
					}
				}
				SOV = {
					transfer_state = PREV
				}
			}

					has_state_flag = slavic_states
				}

			custom_effect_tooltip = SOV_game_totally_over_TT
		}
	}

	option = { #Refuse
		trigger = {
			is_subject = no
		}
		ai_chance = {
			factor = 25
			modifier = {
				factor = 10

			modifier = {
				factor = 75

					is_major = yes

						is_major = yes

					}

						is_major = yes
						has_guaranteed = ROOT
					}
				}
			}
		}
		name = generics.15.b
		if = {
			limit = {
				FROM = {
					is_subject_of = ROOT
				}
			}
			custom_effect_tooltip = FROM_is_puppet_slavic_states

				country_event = { id = generics.19 days = 1 }
			}
		}
		else = {
			SOV = {
				country_event = { id = generics.17 days = 1 }
			}
		}
	}

	option = { #We should ask our master first
