country_event = {
	id = chinas.1
	title = chinas.1.t
	desc = {
		text = chinas.1.d_1
		trigger = {
			JAP = {
				NOT = {
					has_war_with = CHI
				}
			}
		}
	}
	desc = {
		text = chinas.1.d_2
		trigger = {
			JAP = {
				has_war_with = CHI
			}
		}
	} 
	picture = GFX_report_event_chinese_soldiers_running

	is_triggered_only = yes

	option = { # Propose to Japan to Invade China
		name = chinas.1.a
		trigger = {
			NOT = {
				JAP = {
					has_war_with = CHI
				}
			}
			country_exists = CHI
		}
		ai_chance = {
			factor = 60
		}
		JAP = {
			country_event = { id = japans.18 hours = 12 }
		}
	}

	option = { #We Want to Help our Ally JAP - when JAP is in War Against China
		name = chinas.1.b
		trigger = {
			JAP = {
				has_war_with = CHI
			}
			country_exists = CHI
		}
		ai_chance = {
			factor = 60
			modifier = {
				factor = 10
				CHI = {
					is_in_faction_with = ENG
				}
			}
		}
		JAP = {
			country_event = { id = japans.19 hours = 12 }
		}
	}

	option = { #We Should Try to Appease the Situation - Trying to create Peace Between China and Jap
		name = chinas.1.c
		trigger = {
			country_exists = CHI
		}
		ai_chance = {
			factor = 40
			modifier = {
				factor = 0
				CHI = {
					surrender_progress > 0.75
				}
			}
		}
		CHI = {
			country_event = { id = chinas.6 hours = 12 }
		}
		custom_effect_tooltip = sino_japanese_negotiation_surrender
	}

	option = { #Do Nothing
		name = chinas.1.e
		ai_chance = {
			factor = 10
		}
		if = {
			limit = {
				NOT = {
					country_exists = CHI
				}
			}
			custom_effect_tooltip = china_not_exist_because_dead
		}
	}
}
