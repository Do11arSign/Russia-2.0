#Must Give 103 to Italy
country_event = {
	id = italys.4
	title = italys.4.t
	desc = italys.4.d
	picture = GFX_report_event_degaulle_churchill

	fire_only_once = yes

	trigger = {
		controls_state = 103
		OR = {
			is_in_faction_with = SOV
			tag = SOV
			is_subject_of = SOV
		}
		NOT = {
			tag = ITA
		}
		ITA = {
			is_in_faction_with = SOV
			has_country_flag = accepted_russian_deal
			NOT = {
				owns_state = 103
			}
		}
	}

	option = {
		ai_chance = {
			factor = 100
		}
		name = italys.4.a
		ITA = {
			transfer_state = 103
			country_event = { id = italys.5 days = 1 }
		}
	}
}

# Ask Italy to Back Off the Slavic States
country_event = {
	id = italys.6
	title = italys.6.t
	desc = italys.6.d
	picture = GFX_report_event_stalin_meeting

	is_triggered_only = yes

	option = {
		ai_chance = {
			factor = 60
			modifier = {
				has_war = yes
				factor = 90
			}
		}
		name = italys.1.a
		diplomatic_relation = {
			country = SOV
			relation = non_aggression_pact
			active = yes
		}
		every_state = {
			limit = {
				has_state_flag = slavic_states
				is_claimed_by = ITA
			}
			remove_claim_by = ITA
		}
		every_state = {
			limit = {
				has_state_flag = slavic_states
				is_controlled_by = ITA
			}
			SOV = {
				transfer_state = PREV
			}
		}
		SOV = {
			country_event = { id = italys.7 days = 1 }
		}
	}

	option = {
		ai_chance = {
			factor = 40
			modifier = {
				has_war = yes
				factor = 10
			}
		}
		name = italys.1.b
		SOV = {
			country_event = { id = italys.8 days = 1 }
		}
	}
}
